package scalala;
package tensor;
package generic;

/**
 * Builds a Tensor of type To after being given a series of
 * key, value pairs.
 *
 * @author dramage
 */
trait TensorBuilder[@specialized(Int,Long) K, @specialized(Int,Long,Float,Double) V,+To] {
  def update(key : K, value : V);

  def result : To;
}
