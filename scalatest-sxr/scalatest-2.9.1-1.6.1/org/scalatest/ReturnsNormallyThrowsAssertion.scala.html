package org.scalatest

trait ReturnsNormallyThrowsAssertion {

  def returnsNormally(f: => Unit): Boolean = {
    try {
      f
      true
    }
    catch {
      case e: Throwable => false
    }
  }

  def throwsTestFailedException(f: => Unit): Boolean = {
    try {
      f
      false
    }
    catch {
      case e: TestFailedException => true
    }
  }
}
