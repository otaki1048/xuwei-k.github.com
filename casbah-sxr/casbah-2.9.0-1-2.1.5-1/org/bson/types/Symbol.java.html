package org.bson.types;

/**
 *  Class to hold a BSON symbol object, which is an interned string in Ruby
 */
public class Symbol {

    public Symbol(String s) {
        _symbol = s;
    }

    public String getSymbol(){
        return _symbol;
    }

    public String toString(){
        return _symbol;
    }

    private final String _symbol;
}
