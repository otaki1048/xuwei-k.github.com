package net.liftweb
package builtin
package snippet

import scala.xml._
import net.liftweb.http._

object SkipDocType extends DispatchSnippet {

  def dispatch : DispatchIt = {
    case _ => render _
  }

  def render(kids: NodeSeq): NodeSeq = {
    S.skipDocType = true
    kids
  }
}
