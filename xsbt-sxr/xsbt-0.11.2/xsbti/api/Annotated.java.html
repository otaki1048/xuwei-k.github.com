package xsbti.api;

import java.util.Arrays;
import java.util.List;
public class Annotated extends Type 
{
	public Annotated(SimpleType baseType, Annotation[] annotations)
	{
		this.baseType = baseType;
		this.annotations = annotations;
	}
	private final SimpleType baseType;
	private final Annotation[] annotations;
	public final SimpleType baseType()
	{
		return baseType;
	}
	public final Annotation[] annotations()
	{
		return annotations;
	}
	public String toString()
	{
		return "Annotated(" + "baseType: " + baseType() + ", " + "annotations: " + Arrays.toString(annotations())+ ")";
	}
}
