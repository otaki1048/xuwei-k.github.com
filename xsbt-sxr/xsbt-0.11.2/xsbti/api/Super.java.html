public class Super extends PathComponent 
{
	public Super(Path qualifier)
	{
		this.qualifier = qualifier;
	}
	private final Path qualifier;
	public final Path qualifier()
	{
		return qualifier;
	}

	{
		return "Super(" + "qualifier: " + qualifier()+ ")";
	}
}
